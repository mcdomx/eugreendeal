41def _build_gdf_from_image(image, x_range, y_range, filter_val): 

42 """ 

43 Returns a Geopandas data frame 

44 :param image: A 2-dimensional vector of pollution readings data 

45 :param x_range: the longitude range of the map to be generated 

46 :param y_range: the latitude range of the map to be generated 

47 :param filter_val: Readings that should be filtered out (eg types of pollutants) 

48 :return: A Geopandas object with specified readings filtered out 

49 """ 

50 

51 lats = np.linspace(y_range[0], y_range[1], image.shape[0]) 

52 lons = np.linspace(x_range[0], x_range[1], image.shape[1]) 

53 

54 i_df = pd.DataFrame(image, columns=lons, index=lats[::-1]) 

55 lats_list = np.array([i_df.index] * image.shape[1]).flatten() 

56 i_df = i_df.melt(value_vars=i_df.columns).set_index(lats_list).reset_index().rename \ 

57 (columns={'index': 'latitude', 'variable': 'longitude'}) 

58 i_gdf = gpd.GeoDataFrame(i_df, geometry=gpd.points_from_xy(i_df.longitude, i_df.latitude), crs=3857) 

59 

60 i_gdf_filtered = i_gdf[i_gdf.value >= filter_val] 

61 

62 return i_gdf_filtered
