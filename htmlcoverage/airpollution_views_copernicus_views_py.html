41def _convert_imgarray_to_inmem_base64_png(img_array: np.array) -> base64: 

42 """ 

43 ref: https://stackoverflow.com/questions/42503995/how-to-get-a-pil-image-as-a-base64-encoded-string 

44 Convert an image array into an in-memory base64 PNG image. 

45 The return value can be placed in an HTML src tag: 

46 <img src="data:image/png;base64,<<base64 encoding>>" height="" width="" alt="image"> 

47 :param img_array: a numpy image 

48 :return: base64 image in ascii characters 

49 """ 

50 

51 pngimg = Image.new(mode='P', size=img_array.T.shape) # create new in-memory image 

52 # pngimg.putpalette([ 

53 # 255, 255, 255, # white background 

54 # 50, 100, 50, # index 1 is green 

55 # 255, 255, 0, # index 2 is yellow 

56 # 200, 0, 0, # index 3 is red 

57 # ]) 

58 pngimg.putdata([pixel for pixel in np.ravel(img_array)]) # scale=10 <- will multiple each pixel value 

59 

60 b = io.BytesIO() # create an empty byte object 

61 pngimg.save(b, format="PNG") # save the png file to the object 

62 b.seek(0) # move pointer back to the start of memory space 

63 img_bytes = b.read() # read memory space 

64 

65 base64_img = base64.b64encode(img_bytes) # encode the image to base64 

66 base64_ascii_img = base64_img.decode('ascii') # finally, decode it to ascii characters 

67 

68 return base64_ascii_img
