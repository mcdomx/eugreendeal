54 def load_data(self, **kwargs) -> None: 

55 """ 

56 Load RDF data from EEA data source 

57 :param kwargs: kwargs. 

58 """ 

59 

60 # getting the RDF file 

61 target_dir = kwargs.get('target_dir') 

62 baseurl = "http://r.eionet.europa.eu/rdfdumps/clrtap_nec_unfccc/" 

63 filename = "clrtap_nfr09_gf.rdf.gz" 

64 outfilepath = filename[:-3] 

65 

66 response = urllib.request.urlopen(baseurl + filename) 

67 with open(outfilepath, 'wb') as outfile: 

68 outfile.write(gzip.decompress(response.read())) 

69 

70 # creating RDF graph 

71 pollution_graph = rdflib.Graph() 

72 pollution_graph.parse(file=open(outfilepath, 'r'), format='xml') 

73 

74 # performing SPARQL query 

75 results = pollution_graph.query(""" 

76 PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#> 

77 PREFIX clrtap_nec_unfccc: <http://reference.eionet.europa.eu/clrtap_nec_unfccc/schema/> 

78 PREFIX country: <http://reference.eionet.europa.eu/clrtap_nec_unfccc/country/> 

79 PREFIX sector: <http://reference.eionet.europa.eu/clrtap_nec_unfccc/clrtap_nec_nfr09_sector/>  

80 

81 SELECT 

82 ?pollutant_name 

83 ?country_code 

84 ?year 

85 ?sector_code 

86 ?unit 

87 ?emissions 

88 

89 WHERE {{  

90 

91 #?clrtap_nfr09_gf clrtap_nec_unfccc:country_code ?country_code . 

92 #?clrtap_nfr09_gf clrtap_nec_unfccc:pollutant_name ?pollutant_name . 

93 #?clrtap_nfr09_gf clrtap_nec_unfccc:year ?year . 

94 ?clrtap_nfr09_gf clrtap_nec_unfccc:country_code ?country_code . 

95 ?clrtap_nfr09_gf clrtap_nec_unfccc:pollutant_name ?pollutant_name . 

96 ?clrtap_nfr09_gf clrtap_nec_unfccc:sector_code ?sector_code . 

97 ?clrtap_nfr09_gf clrtap_nec_unfccc:unit ?unit . 

98 ?clrtap_nfr09_gf clrtap_nec_unfccc:year ?year . 

99 ?clrtap_nfr09_gf clrtap_nec_unfccc:emissions ?emissions .  

100 

101 FILTER (?year = '2017') 

102 FILTER (?pollutant_name in ('SOx', 'NOx', 'NMVOC', 'CO')) 

103 #to be changed with EEA33 

104 FILTER (?country_code in (country:EU28, country:TR, country:IS, country:NO, country:LI, country:CH)) 

105 #FILTER (?country_code in (country:EEA33)) 

106 

107 } 

108 } 

109 

110 """) 

111 

112 # create a full path to save file to 

113 fpath = os.path.join(target_dir, "EEA_RDF_Complete_DATA.csv") 

114 

115 with open(fpath, 'w', newline='') as file: 

116 writer = csv.writer(file) 

117 for row in results: 

118 p = str(row.asdict()['pollutant_name'].toPython()) 

119 y = str(row.asdict()['year'].toPython()) 

120 u = str(row.asdict()['unit'].toPython()) 

121 e = str(row.asdict()['emissions'].toPython()) 

122 c = row.asdict()['country_code'] 

123 s = row.asdict()['sector_code'] 

124 writer.writerow([p, y, ext(c), ext(s), e, u])

133 def load_db_from_file(filepath: str) -> None: 

134 """ 

135 Loads data from file 

136 :param filepath: path of the file to load 

137 """ 

138 BATCH_RECORD_COUNT = 10000 

139 

140 for obj in EEADataModel.objects.all(): 

141 obj.delete() 

142 

143 with open(filepath, "r+b") as f: 

144 map_file = mmap.mmap(f.fileno(), 0) 

145 record_line_number = [] 

146 for i, _ in enumerate(iter(map_file.readline, b"")): 

147 if i == 0 or i % BATCH_RECORD_COUNT == 0: 

148 record_line_number.append((i, BATCH_RECORD_COUNT, filepath)) 

149 

150 ## Serial processing 

151 for data in record_line_number: 

152 process_file_chunk(data)

155def process_file_chunk(data: tuple): 

156 """ 

157 This is for batch loading 

158 :param data: data to load 

159 """ 

160 logging.info('Received data {} for processing'.format(str(data))) 

161 filepath = data[2] 

162 start_line_num = data[0] 

163 end_line_num = start_line_num + data[1] 

164 

165 with open(filepath) as csv_file: 

166 csv_reader = csv.reader(csv_file, delimiter=',') 

167 records = [] 

168 for i, row in enumerate(csv_reader): 

169 if i == 0 or i < start_line_num: 

170 continue 

171 if i >= end_line_num: 

172 break 

173 

174 try: 

175 emission_value = float(row[4]) 

176 except: 

177 logging.error('Error in converting to float. {}'.format(row)) 

178 continue 

179 

180 record = EEADataModel.objects.create(year=row[1], 

181 pollutant_name=row[0], 

182 unit=row[5], 

183 country=row[2], 

184 sector=row[3], 

185 sector_group=get_sector_group(row[3]), 

186 emissions=emission_value) 

187 records.append(record) 

188 

189 logging.info('Processing lines {} to {}'.format(start_line_num, i - 1)) 

190 with transaction.atomic(): 

191 EEADataModel.objects.bulk_create(records, ignore_conflicts=True) 

192 logging.info('Processed lines {} to {}'.format(start_line_num, i - 1))

199def get_sector_group(sector_code) -> str: 

200 """ 

201 Gets sector group information 

202 :param sector_code: sector code 

203 :return: returns the sector group information 

204 """ 

205 if sector_code in sectors_map["Agriculture"]: 

206 return "Agriculture" 

207 elif sector_code in sectors_map["Commercial, institutional and households"]: 

208 return "Commercial, institutional and households" 

209 elif sector_code in sectors_map["Energy production and distribution"]: 

210 return "Energy production and distribution" 

211 elif sector_code in sectors_map["Energy use in industry"]: 

212 return "Energy use in industry" 

213 elif sector_code in sectors_map["Industrial processes and product use"]: 

214 return "Industrial processes and product use" 

215 elif sector_code in sectors_map["Road transport"]: 

216 return "Road transport" 

217 elif sector_code in sectors_map["Non-road transport"]: 

218 return "Non-road transport" 

219 elif sector_code in sectors_map["Other"]: 

220 return "Other" 

221 elif sector_code in sectors_map["Waste"]: 

222 return "Waste" 

223 else: 

224 return "NATIONAL_TOTAL"

227def extract_name(uri) -> str: 

228 """ 

229 Parse the trailing text from a uri. 

230 If argument is not a uri, returns the initial value. 

231 :param uri: uri to extract name 

232 :return: returns the extracted name 

233 """ 

234 

235 if type(uri) == rdflib.term.URIRef: 

236 uri = uri.toPython() 

237 

238 if uri[-1] != '/': 

239 uri = uri.split('/')[-1] 

240 

241 if uri[-1] != '#': 

242 uri = uri.split('#')[-1] 

243 

244 return uri
