25 def load_data(self, dummy: bool = False, **kwargs) -> dict: 

26 """ 

27 Loads a year at a time only. Will first delete all entries for the year 

28 and country if they already exist. Year_from and Year_to are ignore if since_date provided. 

29 :param dummy: Boolean. If True, limited random records are loaded. 

30 :param kwargs: Supports, year_from:str, year_to:str, since_date:str, country_codes:list 

31 :return: If error, return dict with error description. 

32 """ 

33 # read and load one year at a time or days since a since_date 

34 year_from = int(kwargs.get('year_from', 0)) 

35 year_to = int(kwargs.get('year_to', 0)) 

36 since_date = kwargs.get('since_date', '') # YYYY-MM-DD 

37 country_codes = kwargs.get('country_codes', ['']) 

38 p_list = kwargs.get('pollutants', '') 

39 stations = kwargs.get('stations', '') 

40 month = kwargs.get('month', 0) 

41 

42 # check that either from/to years are provided OR a since_date 

43 # since date takes precendence if all are supplied 

44 if since_date == '' and year_from == '' and year_to == '': 

45 err_text = f"Either from/to years OR since_date argument must be provided." 

46 self.logger.error(err_text) 

47 return {'error': err_text} 

48 

49 if since_date is not None and since_date != '': 

50 year_from = int(since_date.split('-')[0]) 

51 year_to = int(since_date.split('-')[0]) 

52 self.logger.info( 

53 f"Loading observation station readings since {since_date}") 

54 elif month != 0: 

55 self.logger.info(f"Loading observation station readings .. for month:{month} year:{year_to}") 

56 else: 

57 self.logger.info(f"Loading observation station readings .. from {year_from} to {year_to}") 

58 

59 # if no countries provided, select EU countries only 

60 if len(country_codes) == 0: 

61 country_codes = EU_ISOCODES 

62 

63 # get a list of pollutant key strings 

64 # if no specific pollutants provided, get all that are supported 

65 # this retrieves a list of the pollutant keys used for the EEA rest request 

66 if p_list is '': 

67 # ['PM2.5', 'PM10', 'O3', 'NO2', 'CO', 'SO2'] 

68 pollutants = Pollutant.get_observation_pollutants().keys() 

69 

70 # a pollutant list was provided - get the supported pollutants in the EEA key format 

71 else: 

72 try: 

73 p_objs = Pollutant.objects.filter(key__in=p_list) 

74 pollutants = [x.observation_key for x in p_objs] 

75 except Exception as e: 

76 self.logger.info(f"Pollutants not supported: {p_list}") 

77 return {"error": f"Pollutants not supported: {p_list}"} 

78 

79 # if stations are provided, load by stations 

80 if stations != '': 

81 self._load_by_stations(stations, pollutants, year_from, year_to, since_date, month) 

82 

83 # if stations are not provided, use countries 

84 else: 

85 self._load_by_countries(country_codes, pollutants, year_from, year_to, since_date, month) 

86 

87 self.logger.info("Done loading Observation Readings.")

114 def _load_by_stations(self, stations, pollutants, year_from, year_to, since_date, month):

116 err_list = []

119 for p in tqdm(pollutants, desc=f'loading:{c}:{s}', leave=False): 

120 file_list = self._get_reading_filenames(year_from=year_from, 

121 year_to=year_to, 

122 since_date=since_date, 

123 country_code=c, 

124 pollutant=p,

126 

127 if type(file_list) is dict: 

128 e_msg = f"No files: {year_from}-{year_to}:{c}:{p}:{s} : {file_list.get('error', 'ERROR')}" 

129 err_list.append(e_msg) 

130 continue 

131 

132 self._load_filelist(file_list, c, p, s, year_from, year_to, month)

156 """ Load data from a filelist """ 

157 for f in tqdm(file_list, desc=f'files {c}:{p}:{s}:{year_from}-{year_to}', leave=False): 

158 if f == '': 

159 continue 

160 

161 df = self._get_df_from_file(f) 

162 

163 # if month was selected, limit table to the month 

164 if month != 0: 

165 from_date = datetime.datetime(year=year_from, month=month, day=1, tzinfo=pytz.timezone("CET")) 

166 to_date = datetime.datetime(year=year_from, month=month+1, day=1, tzinfo=pytz.timezone("CET")) 

167 df = df[(df['date_time'] >= from_date) & (df['date_time'] < to_date)]

169 if df is not None:

175 def _load_db_from_df(self, df: pd.DataFrame): 

176 """ 

177 Load readings data into database 

178 :param df: dataframe that includes the readings to load 

179 :return: None 

180 """ 

181 # erase everything first 

182 # for station in ObservationStationReading.objects.all(): 

183 # station.delete() 

184 

185 country_lookup = EUCountries.get_country_code_lookup() 

186 pollutant_lookup = Pollutant.get_observation_pollutants() 

187 

188 r_list = [] 

189 for row in tqdm(df.itertuples(), desc='loading rows', total=len(df), leave=False): 

190 

191 # skip invalid readings 

192 if row.Validity != 1: 

193 continue 

194 

195 country_code = country_lookup.get(row.Countrycode) 

196 if country_code is None: 

197 self.logger.debug(f"{row.Countrycode} skipped. Not in NUTS regions.") 

198 continue 

199 

200 pollutant = pollutant_lookup.get(row.AirPollutant) 

201 if pollutant is None: 

202 continue 

203 try: 

204 station = ObservationStation.objects.get(air_quality_station=row.AirQualityStation) 

205 except ObjectDoesNotExist as e: 

206 self.logger.debug(f"Station {row.AirQualityStation} not in master data. Skipping.")

209 try: 

210 r_list.append(ObservationStationReading( 

211 key=row.key, 

212 date_time=row.date_time, 

213 country_code=country_code, 

214 air_quality_network=row.AirQualityNetwork, 

215 air_quality_station=station, 

216 pollutant=pollutant, 

217 value=row.Concentration, 

218 unit=row.UnitOfMeasurement, 

219 validity=row.Validity, 

220 verification=row.Verification)) 

221 

222 except Exception as e: 

223 continue

229 def parse_date(s: pd.Series) -> datetime: 

230 tz = pytz.timezone("CET") 

231 ds = s.DatetimeEnd 

232 dt, tm, _ = ds.split(' ') 

233 yr, mon, day = dt.split('-') 

234 hr, minute, sec = tm.split(':') 

235 return datetime.datetime(int(yr), int(mon), int(day), int(hr), int(minute), int(sec), tzinfo=tz) 

236 

237 def create_key(s: pd.Series) -> str: 

238 return s.date_time.strftime(f"%Y%m%d_%H_{s.AirQualityStation}_{s.AirPollutant}") 

239 

240 try: 

241 df = pd.read_csv(file_name) 

242 except Exception as e: 

243 self.logger.info(file_name, e) 

244 # TODO: for SSL problems run "/Applications/Python\ 3.7/Install\ Certificates.command" 

245 return None 

246 

247 # insert datetime and key fields 

248 df.insert(0, 'date_time', df.apply(parse_date, axis=1)) 

249 df.insert(0, 'key', df.apply(create_key, axis=1)) 

250 

251 return df

253 def _get_reading_filenames(self, 

254 year_from: int, 

255 year_to: int, 

256 since_date: str, 

257 country_code: str, 

258 pollutant: str, 

259 station: str): 

260 

261 # get station readings 

262 url = f'https://fme.discomap.eea.europa.eu/fmedatastreaming/AirQualityDownload/AQData_Extract.fmw?CountryCode={country_code}&CityName=&Pollutant={pollutant}&Year_from={year_from}&Year_to={year_to}&Station={station}&Samplingpoint=&Source=All&Output=TEXT&UpdateDate={since_date}&TimeCoverage=Year' 

263 

264 self.logger.debug("\tgetting request from url:") 

265 self.logger.debug("\t" + url) 

266 f = requests.get(url) 

267 self.logger.debug("\trequest complete.") 

268 

269 if f.status_code != 200: 

270 return {'error': self._handle_web_error(f.status_code)} 

271 

272 self.logger.debug("\tprocessing response ...") 

273 response = f.text 

274 response = response.split('\r') 

275 

276 filelist = [] 

277 for r in response: 

278 filelist.append("".join([x for x in r if x in string.printable]).strip()) 

279 

280 self.logger.debug("\tdone processing response.") 

281 

282 filelist.remove('') 

283 

284 return filelist
