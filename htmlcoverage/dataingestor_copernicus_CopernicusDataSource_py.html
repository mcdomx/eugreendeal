87 def set_token(self) -> str: 

88 """ 

89 Generates a token based on userid and password from the Copernicus API service. 

90 This does not validate the userid/password combination. 

91 Any environment variable stored in COPERNICUS_ID and COPERNICUS_PW will be used to 

92 retrieve token. A token 

93 :return: token string or None if error 

94 """ 

95 

96 t = os.environ.get("COPERNICUS_TOKEN") 

97 # if token is stored, return it 

98 if t: 

99 self._token = t 

100 return self._token 

101 

102 # if user id and pw are available 

103 uid = os.environ.get("COPERNICUS_ID") 

104 pw = os.environ.get("COPERNICUS_PW") 

105 

106 if not uid: 

107 print("A copernicus ID is required to download satellite data.") 

108 print("https://ads.atmosphere.copernicus.eu/user/register") 

109 uid = input("Copernicus User ID: ") 

110 os.environ.setdefault("COPERNICUS_ID", uid) 

111 if not pw: 

112 pw = getpass.getpass() 

113 os.environ.setdefault("COPERNICUS_PW", pw) 

114 

115 url = f'https://geoservices.regional.atmosphere.copernicus.eu/services/GetAPIKey?username={uid}&password={pw}' 

116 token_response = requests.get(url) 

117 

118 if token_response.status_code == 200: 

119 t = BeautifulSoup(token_response.text, 'html.parser').text 

120 self._token = t 

121 return self._token 

122 

123 self.logger.error(f"No token returned: {token_response.status_code}") 

124 return ""

129 def _load_images_direct_to_db(self, reference_time: str) -> str: 

130 """ 

131 Loads satellite image directly into the database. 

132 Only includes an average image for an entire day for pollutants with calendar year targets. 

133 :param reference_time: Time (format: YYYY-MM-DDT00:00:00Z) of the satellite images to load 

134 :return: None 

135 """ 

136 

137 # These must be fixed to get a daily average at the surface 

138 category = 'ANALYSIS' 

139 levels = 'SURFACE' 

140 pollutants = [x.get('pollutant_id') for x in Target.objects.all().values() if 

141 x.get('measurement_id') == 'calendar_year'] 

142 

143 count = 0 

144 for p in pollutants: 

145 url = self._create_file_url(token=self._token, 

146 pollutant_key=p, 

147 reference_time=reference_time, 

148 category=category, 

149 levels=levels) 

150 

151 rv = self._load_netcdf_record_from_url(url) 

152 count += rv 

153 

154 return f"Loaded {count} satellite image records."

156 def _load_netcdf_record_from_url(self, url: str) -> int: 

157 """ 

158 Load the netcdf image and related data diretcly from the url. 

159 :param url: URL to eea data service for the desired image. 

160 :return: Number of records loaded. 

161 """ 

162 

163 self.logger.debug(f"Getting request from Copernicus ... ") 

164 r = requests.get(url) 

165 self.logger.debug(f"Request returned with status {r.status_code}") 

166 

167 if r.status_code != 200: 

168 self.logger.error(f"{self._handle_web_error(r.status_code)}") 

169 self.logger.error(url) 

170 return 0 

171 else: 

172 try: 

173 # determine the filename 

174 fname = self._get_filename_from_cd(r.headers.get('content-disposition')) 

175 if fname is None: 

176 yesterday = (datetime.datetime.today() - datetime.timedelta(1)).isoformat().split('T')[0] 

177 fname = f"{yesterday}Z.nc" 

178 

179 except Exception as e: 

180 print(f'Unable to get filename from url request. {e}') 

181 return 0 

182 

183 # get parameters from the filename 

184 params = self.get_params_from_filepath(fname) 

185 

186 # get in memory netcdf file object 

187 f = io.BytesIO(r.content) 

188 ds = netcdf_file(f) 

189 f.close() 

190 

191 # determine pollutant in the file 

192 cop_pollutant = list(ds.variables.keys())[-1] 

193 pollutant_key = Pollutant.get(cop_pollutant) 

194 

195 # determine date of the file 

196 year, month, day = params['year'], params['month'], params['day'] 

197 

198 # calculate the day average image 

199 surface_images = ds.variables.get(cop_pollutant).data[:, 0, :] 

200 surface_dayavg_image = surface_images.mean(axis=0) 

201 

202 # determine record key from filename 

203 key = fname.split('/')[-1] 

204 

205 # load data for record 

206 load_series = pd.Series({ 

207 'key': key, 

208 'date': f"{year}-{month:2}-{day:2}", 

209 'date_time': datetime.datetime(year=int(year), month=int(month), day=int(day), tzinfo=pytz.timezone("CET")), 

210 'pollutant': Pollutant.objects.get(pk=pollutant_key), 

211 'description': ds.title.decode(), 

212 'model': params['model'], 

213 'category': params['category'], 

214 'bbox_minlon': ds.variables.get('longitude').data.min(), 

215 'bbox_maxlon': ds.variables.get('longitude').data.max(), 

216 'bbox_minlat': ds.variables.get('latitude').data.min(), 

217 'bbox_maxlat': ds.variables.get('latitude').data.max(), 

218 'levels': [int(x) for x in ds.variables.get('level').data], 

219 'hours': [int(x) for x in ds.variables.get('time').data], 

220 'year': year, 

221 'month': month, 

222 'day': day, 

223 'image_shape': surface_dayavg_image.shape, 

224 'file_path': fname, 

225 'image': surface_dayavg_image.flatten() 

226 }) 

227 

228 try: 

229 img = " ".join([str(x) for x in load_series.image.flatten()]) 

230 record = SatelliteImageFiles.objects.create(key=load_series.key, 

231 date=load_series.date, 

232 date_time=load_series.date_time, 

233 pollutant=load_series.pollutant, 

234 description=load_series.description, 

235 model=load_series.model, 

236 category=load_series.category, 

237 bbox_minlon=load_series.bbox_minlon, 

238 bbox_maxlon=load_series.bbox_maxlon, 

239 bbox_minlat=load_series.bbox_minlat, 

240 bbox_maxlat=load_series.bbox_maxlat, 

241 levels=load_series.levels, 

242 hours=load_series.hours, 

243 year=load_series.year, 

244 month=load_series.month, 

245 day=load_series.day, 

246 shape=" ".join([str(x) for x in load_series.image_shape]), 

247 file_path=load_series.file_path, 

248 image=img 

249 ) 

250 record.save() 

251 return 1 

252 

253 except Exception as e: 

254 self.logger.error(f"{e}") 

255 return 0

334 def _create_file_url(self, token: str, 

335 pollutant_key: str, 

336 reference_time: str, 

337 category: str, 

338 levels: str, 

339 times: str = None, 

340 grid_size='0.1', 

341 model='ENSEMBLE', 

342 base_url='https://download.regional.atmosphere.copernicus.eu/services/CAMS50', 

343 fileformat='NETCDF' 

344 ) -> str: 

345 """ 

346 Create an API URL based on arguments. 

347 :param token:ste - A Copernicus token. 

348 :param pollutant_key:str - A pollutant. One of O3, CO, NO2, SO2 ,PM25 , PM10, PANS, NMVOC,NO,NH3. 

349 :param reference_time:str - The day from which the file URL will get images for. Format YYYY-MM-DDT00:00:00Z. If blank, prior day's date is used. 

350 :param base_url:str - The base URL for the API call to download files with images. 

351 :param grid_size:str - Default=0.1. The resolution of the resulting image map. Not tested for values other than 0.1. 

352 :param model:str - The Copernicus model used. One of ENSEMBLE, CHIMERE, EMEP, EURAD-IM, LOTOS-EUROS, MATCH, MOCAGE, SILAM. 

353 :param category:str - Forecast or actual data. FORECAST | ANALYSIS. 

354 :param levels:str - Surface or air levels. One of SURFACE | ALLLEVELS 

355 :param times:str - Range of hours for maps returned in files. For ANALYSIS -24H-1H for forecast 0H24H | 25H48H | 49H72H |73H96H 

356 :param fileformat:str - Format of the returned file. NETCDF | GRIB2 

357 """ 

358 

359 if not reference_time: 

360 yesterday = (datetime.datetime.today() - datetime.timedelta(1)).isoformat().split('T')[0] 

361 reference_time = f"{yesterday}T00:00:00Z" 

362 

363 if not times: 

364 times = '-24H-1H' if category == 'ANALYSIS' else '0H24H' 

365 

366 # check that times are reasonable for forecast cateogry 

367 if category == 'FORECAST': 

368 if times not in ['0H24H', '25H48H', '49H72H', '73H96H']: 

369 self.logger.error( 

370 f"Forecast times of '{times}' not supported. Select one of '0H24H', '25H48H', '49H72H', '73H96H'.") 

371 return None 

372 

373 url = f'{base_url}?token={token}&grid={grid_size}&model={model}&package={category}_{pollutant_key}_{levels}&time={times}&referencetime={reference_time}&format={fileformat}' 

374 

375 return url

507 def get_params_from_filepath(filepath) -> dict: 

508 """ 

509 Return a dictionary of parameters from a netcdf filename. 

510 :param filepath - full path of netcdf file which the file's name will be parsed. 

511 """ 

512 filename = filepath.split('/')[-1] 

513 _, _, details = filename.split(',') 

514 model, cateogry, levels, pollutant, rest = details.split('+') 

515 time_range, _, _, rest = rest.split('_') 

516 date, _ = rest.split('.') 

517 year = date[:4] 

518 month = date[4:6] 

519 day = date[6:8] 

520 hour = date[8:10] 

521 minute = date[10:12] 

522 sec = date[12:] 

523 

524 rv = {'model': model, 

525 'category': cateogry, 

526 'levels': levels, 

527 'pollutant': pollutant, 

528 'timerange': time_range, 

529 'date': date, 

530 'year': year, 

531 'month': month, 

532 'day': day, 

533 'hour': hour, 

534 'minute': minute, 

535 'second': sec} 

536 

537 return rv
